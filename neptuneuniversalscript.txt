local function SaveRegistryToFile()
    local Success, EncodedString = pcall(function()
        return HttpService:JSONEncode(ConfigsRegistry)
    end)
    if Success and writefile then
        writefile(FILE_NAME, EncodedString)
    end
end

local function LoadRegistryFromFile()
    if readfile and isfile and isfile(FILE_NAME) then
        local Success, DecodedTable = pcall(function()
            return HttpService:JSONDecode(readfile(FILE_NAME))
        end)
        if Success and type(DecodedTable) == "table" then
            ConfigsRegistry = DecodedTable
            if not ConfigsRegistry._GlobalMetadata then
                ConfigsRegistry._GlobalMetadata = { AutoLoadConfig = "" }
            end
        end
    end
end

local function SwitchActiveUITheme(ThemeData, ThemeName)
    ActiveTheme = ThemeData
    if ThemeName then ActiveThemeName = ThemeName end
    
    Main.BackgroundColor3 = ThemeData.MainBg
    Sidebar.BackgroundColor3 = ThemeData.SidebarBg
    MainStroke.Color = ThemeData.Accent
    
    for _, frame in ipairs(ObjectRegistry.Sections) do frame.BackgroundColor3 = ThemeData.SectionBg end
    for _, stroke in ipairs(ObjectRegistry.Strokes) do stroke.Color = ThemeData.Accent end
    for _, btn in ipairs(ObjectRegistry.Buttons) do btn.BackgroundColor3 = ThemeData.BtnBg end
    for _, txt in ipairs(ObjectRegistry.TextAccents) do txt.TextColor3 = ThemeData.TextAccent end
    for _, track in ipairs(ObjectRegistry.SliderTracks) do track.BackgroundColor3 = ThemeData.SidebarBg end
    for _, fill in ipairs(ObjectRegistry.SliderFills) do fill.BackgroundColor3 = ThemeData.Accent end
    for _, sb in ipairs(ObjectRegistry.Sidebars) do sb.BackgroundColor3 = ThemeData.SidebarBg end
    
    for _, tabData in ipairs(ObjectRegistry.TabButtons) do
        if tabData.Active then
            tabData.Instance.BackgroundTransparency = 0.7
            tabData.Instance.TextColor3 = Color3.new(1,1,1)
            tabData.Instance.BackgroundColor3 = ThemeData.Accent
        else
            tabData.Instance.BackgroundTransparency = 1
            tabData.Instance.TextColor3 = Color3.fromRGB(160, 190, 220)
        end
    end
end

            node.Active = false
            node.Instance.BackgroundTransparency = 1
            node.Instance.TextColor3 = Color3.fromRGB(160, 190, 220)

        TabBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
        Page.Visible = true
    end)

    Tabs[name] = {Btn = TabBtn, Page = Page}
    return Page

local function CreateDarkSlider(parent, labelText, min, max, default, layoutOrder, isDecimal, updateCallback)
    local Container = Instance.new("Frame")
    Container.Size = UDim2.new(1, 0, 0, 34)
    Container.BackgroundTransparency = 1
    Container.LayoutOrder = layoutOrder
    Container.Parent = parent

    local Label = Instance.new("TextLabel")
    Label.Size = UDim2.new(1, 0, 0, 16)
    Label.BackgroundTransparency = 1
    Label.Font = Enum.Font.GothamSemibold
    Label.Text = labelText .. ": " .. default
    Label.TextColor3 = Color3.fromRGB(230, 240, 255)
    Label.TextSize = 11
    Label.TextXAlignment = Enum.TextXAlignment.Left
    Label.Parent = Container

    local Track = Instance.new("Frame")
    Track.Size = UDim2.new(1, 0, 0, 8)
    Track.Position = UDim2.new(0, 0, 0, 22)
    Track.BackgroundColor3 = ActiveTheme.SidebarBg
    Track.Parent = Container
    Instance.new("UICorner", Track).CornerRadius = UDim.new(1, 0)
    table.insert(ObjectRegistry.SliderTracks, Track)

    local Fill = Instance.new("Frame")
    Fill.Size = UDim2.new(0, 0, 1, 0)
    Fill.BackgroundColor3 = ActiveTheme.Accent
    Fill.Parent = Track
    Instance.new("UICorner", Fill).CornerRadius = UDim.new(1, 0)
    table.insert(ObjectRegistry.SliderFills, Fill)

    local Knob = Instance.new("Frame")
    Knob.Size = UDim2.new(0, 12, 0, 12)
    Knob.AnchorPoint = Vector2.new(0.5, 0.5)
    Knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    Knob.Parent = Track
    Instance.new("UICorner", Knob).CornerRadius = UDim.new(1, 0)

    local function SetValue(val)
        val = math.clamp(val, min, max)
        local pct = (val - min) / (max - min)
        Fill.Size = UDim2.new(pct, 0, 1, 0)
        Knob.Position = UDim2.new(pct, 0, 0.5, 0)
        
        if isDecimal then
            Label.Text = labelText .. ": " .. string.format("%.2f", val) .. " ms"
        else
            Label.Text = labelText .. ": " .. math.floor(val)
        end
    end

    local Dragging = false
    local function Update(mouseX)
        local pct = math.clamp((mouseX - Track.AbsolutePosition.X) / Track.AbsoluteSize.X, 0, 1)
        local val = min + ((max - min) * pct)
        SetValue(val)
        updateCallback(val)
    end

    Track.InputBegan:Connect(function(inp)
        if inp.UserInputType == Enum.UserInputType.MouseButton1 then Dragging = true; Update(inp.Position.X) end
    end)
    UserInputService.InputChanged:Connect(function(inp)
        if Dragging and inp.UserInputType == Enum.UserInputType.MouseMovement then Update(inp.Position.X) end
    end)
    UserInputService.InputEnded:Connect(function(inp)
        if inp.UserInputType == Enum.UserInputType.MouseButton1 then Dragging = false end
    end)

    SetValue(default)
    return SetValue
end

------------------ PERFORMANCE FLIGHT MECHANICAL RECONSTRUCT ------------------
local function StopFlying()
    if FlyConnection then FlyConnection:Disconnect(); FlyConnection = nil end
    local Character = Player.Character
    local RootPart = Character and Character:FindFirstChild("HumanoidRootPart")
    local Humanoid = Character and Character:FindFirstChildOfClass("Humanoid")
    if Humanoid then 
        Humanoid.PlatformStand = false
        Humanoid:ChangeState(Enum.HumanoidStateType.Running)
    end
    if RootPart then RootPart.AssemblyLinearVelocity = Vector3.new(0, 0.1, 0) end
end

local function StartFlying()
    StopFlying()
    if not Flying then return end
    local Character = Player.Character
    local RootPart = Character and Character:FindFirstChild("HumanoidRootPart")
    local Humanoid = Character and Character:FindFirstChildOfClass("Humanoid")
    if not RootPart or not Humanoid then return end

    FlyConnection = RunService.PreRender:Connect(function(deltaTime)
        if not Flying or not Character or not RootPart or not Humanoid then StopFlying(); return end
        local Camera = workspace.CurrentCamera
        local Direction = Vector3.new(0, 0, 0)

        if UserInputService:IsKeyDown(Enum.KeyCode.W) then Direction = Direction + Camera.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then Direction = Direction - Camera.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then Direction = Direction + Camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then Direction = Direction - Camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then Direction = Direction + Vector3.new(0, 1, 0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then Direction = Direction - Vector3.new(0, 1, 0) end

        if Direction.Magnitude == 0 then
            Humanoid.PlatformStand = false
            Humanoid:ChangeState(Enum.HumanoidStateType.Running)
            RootPart.AssemblyLinearVelocity = Vector3.new(0, -0.1, 0)
        else
            Humanoid.PlatformStand = true
            Humanoid:ChangeState(Enum.HumanoidStateType.Physics)
            local MoveVelocity = Direction.Unit * FlySpeed
            RootPart.AssemblyLinearVelocity = MoveVelocity
            RootPart.CFrame = CFrame.new(RootPart.Position + (MoveVelocity * deltaTime)) * CFrame.Angles(Camera.CFrame:ToEulerAnglesXYZ())
        end
    end)
end

local function RefreshAutocomplete(text)
    for _, child in ipairs(DropdownFrame:GetChildren()) do
        if child:IsA("TextButton") then child:Destroy() end
    end
    
    if text == "" then 
        DropdownFrame.Size = UDim2.new(1, 0, 0, 0)
        return 
    end
    
    local Matches = {}
    local MatchCount = 0
    local SearchQuery = string.lower(text)
    
    for _, p in ipairs(Players:GetPlayers()) do
        if string.find(string.lower(p.Name), SearchQuery, 1, true) then
            table.insert(Matches, p.Name)
            MatchCount = MatchCount + 1
            if MatchCount >= 4 then break end 
        end
    end
    
    if MatchCount > 0 then
        DropdownFrame.Size = UDim2.new(1, 0, 0, (MatchCount * 26) + 4)
        for _, name in ipairs(Matches) do
            local ChoiceBtn = Instance.new("TextButton")
            ChoiceBtn.Size = UDim2.new(1, 0, 0, 24)
            ChoiceBtn.BackgroundColor3 = Color3.fromRGB(15, 36, 60)
            ChoiceBtn.Font = Enum.Font.Gotham
            ChoiceBtn.Text = "  " .. name
            ChoiceBtn.TextColor3 = Color3.fromRGB(220, 235, 255)
            ChoiceBtn.TextSize = 12
            ChoiceBtn.TextXAlignment = Enum.TextXAlignment.Left
            ChoiceBtn.ZIndex = 11
            ChoiceBtn.Parent = DropdownFrame
            
            ChoiceBtn.MouseButton1Click:Connect(function()
                PlayerBox.Text = name
                RefreshAutocomplete("")
            end)
        end
    else
        DropdownFrame.Size = UDim2.new(1, 0, 0, 0)
    end
end

local function GetActualBoneName()
    if TargetBone == "Torso" then return "HumanoidRootPart" end
    if TargetBone == "Feet" then return "LeftFoot" end
    return "Head"
end

local function IsPlayerVisible(targetPart, useTriggerWallcheck)
    local checkValue = useTriggerWallcheck and TriggerWallCheck or WallCheckEnabled
    if not checkValue then return true end
    local Origin = CurrentCamera.CFrame.Position
    local Direction = targetPart.Position - Origin
    
    local RaycastParamsData = RaycastParams.new()
    RaycastParamsData.FilterType = Enum.RaycastFilterType.Exclude
    RaycastParamsData.FilterDescendantsInstances = {Player.Character, targetPart.Parent} 
    RaycastParamsData.IgnoreWater = true
    
    local Result = workspace:Raycast(Origin, Direction, RaycastParamsData)
    return Result == nil 
end

local function FetchClosestPlayerInFov()
    local TargetPlayer = nil
    local MicroDistance = math.huge
    local CenterScreenPos = Vector2.new(CurrentCamera.ViewportSize.X / 2, CurrentCamera.ViewportSize.Y / 2)

    for _, v in ipairs(Players:GetPlayers()) do
        if v ~= Player and v.Character and v.Character:FindFirstChild("Humanoid") and v.Character.Humanoid.Health > 0 then
            local Part = v.Character:FindFirstChild(GetActualBoneName()) or v.Character:FindFirstChild("HumanoidRootPart")
            if Part then
                local ScreenPos, OnScreen = CurrentCamera:WorldToViewportPoint(Part.Position)
                if OnScreen then
                    local ScreenVec = Vector2.new(ScreenPos.X, ScreenPos.Y)
                    local TargetDistance = (ScreenVec - CenterScreenPos).Magnitude
                    if TargetDistance <= FovRadius and TargetDistance < MicroDistance then
                        if IsPlayerVisible(Part, false) then
                            MicroDistance = TargetDistance
                            TargetPlayer = v
                        end
                    end
                end
            end
        end
    end
    return TargetPlayer
end

local IsFiring = false
local function CheckTriggerbotIntersection()
    if not TriggerBotEnabled or IsFiring then return end
    
    local CenterRay = CurrentCamera:ViewportPointToRay(CurrentCamera.ViewportSize.X / 2, CurrentCamera.ViewportSize.Y / 2)
    local RaycastParamsData = RaycastParams.new()
    RaycastParamsData.FilterType = Enum.RaycastFilterType.Exclude
    RaycastParamsData.FilterDescendantsInstances = {Player.Character}
    RaycastParamsData.IgnoreWater = true
    
    local Result = workspace:Raycast(CenterRay.Origin, CenterRay.Direction * 400, RaycastParamsData)
    
    if Result and Result.Instance then
        local HitObject = Result.Instance
        local TargetChar = HitObject.Parent
        
        while TargetChar and TargetChar ~= workspace and not TargetChar:FindFirstChildOfClass("Humanoid") do
            TargetChar = TargetChar.Parent
        end
        
        if TargetChar and TargetChar:IsA("Model") then
            local TargetPlr = Players:GetPlayerFromCharacter(TargetChar)
            local Humanoid = TargetChar:FindFirstChildOfClass("Humanoid")
            
            if TargetPlr and TargetPlr ~= Player and Humanoid and Humanoid.Health > 0 then
                local RootPart = TargetChar:FindFirstChild("HumanoidRootPart")
                if RootPart and IsPlayerVisible(RootPart, true) then
                    IsFiring = true
                    
                    if TriggerDelay > 0 then
                        task.wait(TriggerDelay / 1000)
                    end
                    
                    if mouse1press and mouse1release then
                        mouse1press()
                        task.wait(0.035) 
                        mouse1release()
                    end
                    
                    task.wait(0.05) 
                    IsFiring = false
                end
            end
        end
    end
end

                end
            end
        end
    end
    
    CheckTriggerbotIntersection()

local function ExecuteConfigLoad(data)
    WalkSpeed = data.WalkSpeed or 16
    WalkSpeedSliderRef(WalkSpeed)
    
    Flying = data.Flying or false
    InfiniteJump = data.InfiniteJump or false
    Noclip = data.Noclip or false
    GodMode = data.GodMode or false
    EspEnabled = data.EspEnabled or false
    
    if data.ToggleKeyString then
        ToggleKey = Enum.KeyCode[data.ToggleKeyString]
        BindLabel.Text = "UI Close Bind: " .. ToggleKey.Name
    end
    
    AimlockEnabled = data.AimlockEnabled or false
    TargetBone = data.TargetBone or "Head"
    WallCheckEnabled = data.WallCheckEnabled or false
    ShowFovCircle = data.ShowFovCircle or false
    FovRadius = data.FovRadius or 100
    AimSmoothness = data.AimSmoothness or 0
    
    TriggerBotEnabled = data.TriggerBotEnabled or false
    TriggerWallCheck = data.TriggerWallCheck or false
    TriggerDelay = data.TriggerDelay or 10
    
    if data.FovColorR then FovColor = Color3.new(data.FovColorR, data.FovColorG, data.FovColorB) end
    if data.FillR then EspFillColor = Color3.new(data.FillR, data.FillG, data.FillB) end
    if data.OutR then EspOutlineColor = Color3.new(data.OutR, data.OutG, data.OutB) end
    
    if data.ThemePreset and UIThemeRegistry[data.ThemePreset] then
        SwitchActiveUITheme(UIThemeRegistry[data.ThemePreset], data.ThemePreset)
    end
    
    MenuAnimationType = data.MenuAnimationType or "Pop"
    if CurrentAnimLabelRef then
        CurrentAnimLabelRef.Text = "Current Toggle Animation: " .. string.upper(MenuAnimationType)
    end
    
    UpdateToggleTextFormatting(FlyBtnRef, "Fly HACK", Flying)
    UpdateToggleTextFormatting(JumpBtnRef, "Infinite Jump", InfiniteJump)
    UpdateToggleTextFormatting(NoclipBtnRef, "Noclip Walls", Noclip)
    UpdateToggleTextFormatting(GodBtnRef, "Unlimited Health", GodMode)
    UpdateToggleTextFormatting(EspBtnRef, "Player Highlight ESP", EspEnabled)
    
    UpdateToggleTextFormatting(AimlockBtnRef, "Lock Aim System", AimlockEnabled)
    UpdateBoneButtonText()
    UpdateToggleTextFormatting(WallCheckBtnRef, "Wall Check (Line of Sight)", WallCheckEnabled)
    UpdateToggleTextFormatting(FovShowBtnRef, "Display FOV Ring Overlay", ShowFovCircle)
    
    UpdateToggleTextFormatting(TriggerBtnRef, "Auto Trigger Bot", TriggerBotEnabled)
    UpdateToggleTextFormatting(TriggerWallBtnRef, "Trigger Wallcheck", TriggerWallCheck)
    
    SetFovRangeSliderRef(FovRadius)
    SetSmoothnessSliderRef(AimSmoothness)
    SetTriggerDelaySliderRef(TriggerDelay)
    SetFovColorRef(FovColor)
    SetFillColorRef(EspFillColor)
    SetOutlineColorRef(EspOutlineColor)
    
    UpdateAllEsp()
    if Flying then StartFlying() else StopFlying() end
end

local function RebuildUnifiedList()
    for _, item in ipairs(UnifiedScroll:GetChildren()) do if item:IsA("TextButton") then item:Destroy() end end
    local Count = 0
    
    for cfgName, data in pairs(ConfigsRegistry) do
        if cfgName ~= "_GlobalMetadata" then
            Count = Count + 1
            local ItemBtn = Instance.new("TextButton")
            ItemBtn.Size = UDim2.new(1, -6, 0, 24)
            ItemBtn.Font = Enum.Font.Gotham
            ItemBtn.TextSize = 11
            ItemBtn.TextXAlignment = Enum.TextXAlignment.Left
            ItemBtn.Parent = UnifiedScroll
            Instance.new("UICorner", ItemBtn).CornerRadius = UDim.new(0, 4)

            local prefix = ""
            if ConfigsRegistry._GlobalMetadata.AutoLoadConfig == cfgName then
                prefix = "[AUTO] "
                ItemBtn.BackgroundColor3 = Color3.fromRGB(20, 55, 45)
                ItemBtn.TextColor3 = Color3.fromRGB(120, 255, 180)
            elseif SelectedConfigName == cfgName then
                ItemBtn.BackgroundColor3 = Color3.fromRGB(0, 100, 200)
                ItemBtn.TextColor3 = Color3.new(1,1,1)
            else
                ItemBtn.BackgroundColor3 = Color3.fromRGB(14, 34, 56)
                ItemBtn.TextColor3 = Color3.fromRGB(200, 220, 240)
            end
            
            ItemBtn.Text = "  " .. prefix .. cfgName

            ItemBtn.MouseButton1Click:Connect(function()
                SelectedConfigName = cfgName
                ListLabel.Text = "Selected Config: " .. string.upper(cfgName)
                RebuildUnifiedList()
            end)
        end
    end
    UnifiedScroll.CanvasSize = UDim2.new(0, 0, 0, Count * 28)
end

local ToggleDebounce = false
UserInputService.InputBegan:Connect(function(Input, EngineProcessed)
    if EngineProcessed then return end
    if WaitingForKey and Input.UserInputType == Enum.UserInputType.Keyboard then
        ToggleKey = Input.KeyCode
        WaitingForKey = false
        KeybindButton.Text = "Change Bind Key"
        BindLabel.Text = "UI Close Bind: " .. ToggleKey.Name
        return
    end
    if Input.KeyCode == ToggleKey then
        if ToggleDebounce then return end
        MenuVisible = not MenuVisible
        
        if MenuAnimationType == "None" then
            Main.Visible = MenuVisible
        elseif MenuAnimationType == "Pop" then
            ToggleDebounce = true
            if MenuVisible then
                Main.Visible = true
                MainScale.Scale = 0
                local t = TweenService:Create(MainScale, TweenInfo.new(0.35, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Scale = 1})
                t:Play()
                task.wait(0.35)
            else
                local t = TweenService:Create(MainScale, TweenInfo.new(0.25, Enum.EasingStyle.Back, Enum.EasingDirection.In), {Scale = 0})
                t:Play()
                task.wait(0.25)
                Main.Visible = false
                MainScale.Scale = 1
            end
            ToggleDebounce = false
        elseif MenuAnimationType == "Slide" then
            ToggleDebounce = true
            if MenuVisible then
                Main.Visible = true
                Main.Position = UDim2.new(LastMenuPosition.X.Scale, LastMenuPosition.X.Offset, 1.5, 0)
                local t = TweenService:Create(Main, TweenInfo.new(0.4, Enum.EasingStyle.Exponential, Enum.EasingDirection.Out), {Position = LastMenuPosition})
                t:Play()
                task.wait(0.4)
            else
                LastMenuPosition = Main.Position
                local t = TweenService:Create(Main, TweenInfo.new(0.3, Enum.EasingStyle.Exponential, Enum.EasingDirection.In), {Position = UDim2.new(LastMenuPosition.X.Scale, LastMenuPosition.X.Offset, 1.5, 0)})
                t:Play()
                task.wait(0.3)
                Main.Visible = false
                Main.Position = LastMenuPosition
            end
            ToggleDebounce = false
        end
    end
end)
